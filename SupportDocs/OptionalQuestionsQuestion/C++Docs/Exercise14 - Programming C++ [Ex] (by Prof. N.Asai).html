#include <iostream>
using namespace std;
 
class vec{
int *p, int low, high;
 
public:
vec(int l,int h){
low = l;
high = h;
p = new int[high-low+1];
 
int & operator [] (int i){
return p[i-low];
}
 
};
 
int main(){
 vec<float> v(-10, 20);
 v[-10] = 10.0;
 v[20] = -10.0;
 cout << "v[" << -10 << "]" = << v[-10] << endl;
 cout << "v[" << 20 << "]" = << v[20] << endl;
 cout << "v[" << 30 << "]" = << v[30] << endl;
 cout << "v[" << 40 << "]" = << v[40] << endl;
}

#include <string>

#include <vector>
 
using namespace std;
 
class ticket{
public:
 string ID;
 string seat;
 ticket(): ID("***"),seat("***"){
   cout << "ticket default constructor" << endl;
 }
 ticket(string i, string t): ID(i), seat(t){
   cout << "ticket parameter-constructor" << endl;
 }
 void printInfo(){
  cout << "ID:" << ID << " Seat:" << seat << endl;
 }
};
  
class airplaneTicket: public ticket{
public:
 string flight;
 airplaneTicket():flight("***"){
   cout << "airplane default constructor" << endl;

  cout << "airplane parameter-constructor" << endl;
 }

 }
};
